#include<iostream>
#include<vector>
#include<cctype>
using namespace std;
int main(){
    vector<char>v(10);
    vector<char>::iterator p;
    unsigned int i;
    p=v.begin();
    i=0;
    while(p!=v.end()){
        *p=i+'a';
        p++;
        i++;
    }
    cout<<"original content with Iterator:\n";
    p=v.begin();
    while(p!=v.end()){
        cout<<*p<<" ";
        p++;
    }
    cout<<"\nSize="<<v.size()<<endl;
    for(i=0;i<10;i++){
        v[i]=i+'a';
    }
    cout<<"Current content:\n";
    for(i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
    cout<<"\n\n";
    cout<<"Exppanding vector:\n";
    for(i=0;i<10;i++){
        v.push_back(i+10+'a');
    }
    cout<<"Size now="<<v.size()<<endl;
    cout<<"Current content:\n";
    for(i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
    cout<<"\n\n";
    for(i=0;i<v.size();i++){
        v[i]=toupper(v[i]);
    }
    cout<<"Modified content without Iterator:\n";
    for(i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
    cout<<endl;
    return 0;
}
